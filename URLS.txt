URL api:
    https://epitech-api.herokuapp.com

login: POST
    /login ? "login"=LOGIN & "password"=PASS
        TOKEN <- token

get dashboard: POST
    /infos ? "token"=TOKEN

get planning: GET
    /planning ? "token"=TOKEN & "start"=YYYY-MM-DD & "end"=YYYY-MM-DD

get all projects: GET
    /projects ? "token"=TOKEN

get project: GET
    /project ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI

subscribe project: POST
    /project ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI

unsubscribe project: DELETE
    /project ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI

download project file: GET
    /project/files ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI

get user modules: GET
    /modules ? "token"=TOKEN

get all modules: GET
    /allmodules ? "token"=TOKEN & "scolaryear"=YEAR & "location"=LOCATION & "course"=COURSE

get module: GET
    /module ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE

subscribe module: POST
    /module ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE

unsubscribe module: DELETE
    /module ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE

get event: GET
    /event ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI & "codeevent"=EVENT

subscribe event: POST
    /event ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI & "codeevent"=EVENT

unsubscribe event: DELETE
    /event ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI & "codeevent"=EVENT

get marks: GET
    /marks ? "token"=TOKEN

get messages: GET
    /messages ? "token"=TOKEN

get alerts: GET
    /alerts ? "token"=TOKEN

get photo url: GET
    /photo ? "token"=TOKEN & "login"=LOGIN

validate token: POST
    /token ? "token"=TOKEN & "scolaryear"=YEAR & "codemodule"=MODULE & "codeinstance"=INSTANCE & "codeacti"=ACTI & "codeevent"=EVENT & "tokenvalidationcode"=TOKENTOVALIDATE

get user info: GET
    /user ? "token"=TOKEN & "user"=LOGIN
